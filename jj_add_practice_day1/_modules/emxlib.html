    def getEnviron(cls):
        """ Setup the environment variables needed to launch emx export. """
        environ = Environ(os.environ)

        environ.update({
            'PATH': Plugin.getHome(),
            'LD_LIBRARY_PATH': str.join(cls.getHome(), 'emxlib')
                               + ":" + cls.getHome(),
        }, position=Environ.BEGIN)

        return environ
